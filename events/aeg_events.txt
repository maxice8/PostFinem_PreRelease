namespace = aeg_events

#Opening event in chain
#Introduce dilemma and expands unrest
country_event = {
	id = aeg_events.1
	title = aeg_events.1.t
	desc = aeg_events.1.d
	picture = wihgfx_FEMALE_RULER_eventPicture
	
	trigger = {
		has_country_modifier = aeg_userper_regent_modifier
	}

	fire_only_once = yes
	
	mean_time_to_happen = {
		months = 12
	}
	option = {
		name = aeg_events.1.a #Regent Path
		ai_chance = { factor = 1 }
		add_legitimacy = 25
		custom_tooltip = " "
		define_advisor = {
			type = commandant
			skill = 2
			discount = yes
		}
		custom_tooltip = " "
		add_ruler_modifier = {
			name = aeg_regent_acceptance_modifier
			duration = -1
		}
		set_country_flag = regent_path_flag
	}
	option = {
		name = aeg_events.1.b #Theodoro Path
		ai_chance = { factor = 10 }
		add_stability = -1
		custom_tooltip = " "
		capital_scope = {
			spawn_rebels = {
				size = 3
				type = noble_rebels
				unrest = 5
				win = yes
			}
		}
		random_owned_province = {
			limit = {
				any_neighbor_province = {
					is_capital_of = ROOT
				}
			}
			spawn_rebels = {
				size = 3
				type = noble_rebels
				unrest = 5
				win = yes
			}
		}
		custom_tooltip = " "
		random_owned_province = {
			spawn_rebels = {
				size = 1
				type = particularist_rebels
				unrest = 5
			}
		}
		random_owned_province = {
			spawn_rebels = {
				size = 1
				type = particularist_rebels
				unrest = 5
			}
		}
		custom_tooltip = " "
		add_country_modifier = {
			name = aeg_loyalists_revolt_modifier
			duration = 3650
		}
		set_country_flag = theodoro_path_flag
	}
}

#Isadora has stabilized the realm and is viewed as the rightful leader of Aegyptos, albiet with a weakened heir.
country_event = {
	id = aeg_events.3
	title = aeg_events.3.t
	desc = aeg_events.3.d
	picture = wihgfx_FEMALE_RULER_eventPicture
	
	trigger = {
		has_country_flag = regent_path_flag
		stability = 1
		legitimacy = 85
		has_ruler_flag = isadora_gidos_flag
	}

	fire_only_once = yes
	
	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = aeg_events.3.a #The light at the end of the tunnel
		ai_chance = { factor = 1 }
		add_stability_or_adm_power = yes
		capital_scope = {
			add_base_tax = 2
			add_base_production = 1
			add_base_manpower = 1
		}
		custom_tooltip = " "
		remove_country_modifier = aeg_userper_regent_modifier
	}

}

#Isadora is finally overthrown and Theodoro is installed
country_event = {
	id = aeg_events.5
	title = aeg_events.5.t
	desc = aeg_events.5.d
	picture = wihgfx_FEMALE_RULER_eventPicture
	
	trigger = {
		has_country_flag = theodoro_path_flag
		heir_age = 15
		has_heir_flag = theodoro_gidos_flag
		has_ruler_flag = isadora_gidos_flag
		NOT = { has_spawned_rebels = noble_rebels }
		NOT = { has_spawned_rebels = particularist_rebels }
	}

	fire_only_once = yes
	
	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = aeg_events.5.a #Burn the witch!
		ai_chance = { factor = 1 }
		kill_ruler = yes
		custom_tooltip = " "
		add_stability_or_adm_power = yes
		custom_tooltip = " "
		change_adm = 1
		change_dip = 1
		change_mil = 1
		custom_tooltip = " "
		add_ruler_modifier = {
			name = aeg_theodoro_ascended_modifier
		}
	}

}
