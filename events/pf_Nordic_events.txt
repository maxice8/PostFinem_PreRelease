country_event = {
	id = pf_nordic.13
	title = "pf_nordic.13.t"
	desc = "pf_nordic.13.d"
	picture = BATTLE_eventPicture

	trigger = {
		check_variable = {
			which = RaidLoot
			value = 1
		}
		NOT = {
			is_in_war = {
				casus_belli = cb_raid
				attacker_leader = ROOT
			}
		}
		NOT = {
			any_province = {
				NOT = {
					OR = {
						owned_by = ROOT
						owner = { is_subject_of = ROOT }
					}
				}
				OR = {
					infantry_in_province = ROOT
					cavalry_in_province = ROOT
					artillery_in_province = ROOT
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		set_country_flag = in_raid_event
	}

	after = {
		clr_country_flag = in_raid_event
	}
	option = {		# Collect the loot
		name = "pf_nordic.13.a"
		custom_tooltip = gain_x_loot_tooltip
		hidden_effect = {
			while = {
				limit = {
					check_variable = {
						which = RaidLoot
						value = 1
					}
				}
				subtract_variable = {
					which = RaidLoot
					value = 1
				}
				add_treasury = 1
			}
		}
	}
}
