country_event = {		# Feast Start Screen
	id = pf_feast.1
	title = pf_feast.1.t 
	desc = pf_feast.1.d
	picture = FEAST_eventPicture

	is_triggered_only = yes

	immediate = {
		clr_country_flag = is_in_feast
	}

	option = {
		name = "pf_feast.1.a"
		trigger = {
			government_rank = 1
		}
		highlight = yes
		if = {
			limit = {
				NOT = { has_government_attribute = enables_discounted_host_feast }
			}
			add_years_of_income = -0.5
		}
		else = {
			add_years_of_income = -0.25
		}
		set_country_flag = tier_1_feast
		custom_tooltip = tier_1_feast_tt
		set_country_flag = hosted_feast_flag
		if = {
			limit = {
				has_country_flag = feasts_give_reputation
			}
			add_ruler_modifier = {
				name = "pf_feast_reputation_boost"
				duration = -1
				desc = "until the end of the feast"
			}
		}
		set_country_flag = is_in_feast
		hidden_effect = {
			country_event = {
				id = pf_feast.10
				days = 180
			}
		}
	}
	option = {
		name = "pf_feast.1.b"
		trigger = {
			government_rank = 2
		}
		highlight = yes
		if = {
			limit = {
				NOT = { has_government_attribute = enables_discounted_host_feast }
			}
			add_years_of_income = -2
		}
		else = {
			add_years_of_income = -1
		}
		set_country_flag = tier_2_feast
		custom_tooltip = tier_2_feast_tt
		set_country_flag = hosted_feast_flag
		if = {
			limit = {
				has_country_flag = feasts_give_reputation
			}
			add_ruler_modifier = {
				name = "pf_feast_reputation_boost"
				duration = -1
				desc = "until the end of the feast"
			}
		}
		set_country_flag = is_in_feast
		hidden_effect = {
			country_event = {
				id = pf_feast.10
				days = 365
			}
		}
	}
	option = {
		name = "pf_feast.1.c"
		trigger = {
			government_rank = 3
		}
		highlight = yes
		if = {
			limit = {
				NOT = { has_government_attribute = enables_discounted_host_feast }

		else = {
			add_years_of_income = -2.5

		custom_tooltip = tier_3_feast_tt

country_event = {			# Feast: Nobles Quarrel
	id = pf_feast.2
	title = pf_feast.2.t
	desc = "pf_feast.2.d"
	picture = FEAST_eventPicture

	trigger = {
		has_country_flag = is_in_feast
		has_estate = estate_nobles
	}

	is_triggered_only = yes

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			set_saved_name = {
    				key = noble_1
    				type = advisor
    				scope = ROOT
			}
			set_saved_name = {
    				key = noble_2
    				type = advisor
    				scope = ROOT
			}
		}
	}
	after = {
		clear_saved_name = noble_1
		clear_saved_name = noble_2
	}
	
	option = {		# Reprimand them both
		name = pf_feast.2.a
		add_prestige = 5
		add_legitimacy = 5
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -10
		}
	}
	option = {		# Join the debate
		name = pf_feast.2.b
		add_prestige = -5
		random_list = {
			50 = {
				modifier = {
					factor = 1.2
					dip = 4
				}
				modifier = {
					factor = 1.2
					dip = 5
				}
				modifier = {
					factor = 1.2
					dip = 6
				}
				add_dip_power = 50
			}
			50 = {
				modifier = {
					factor = 1.2
					NOT = { dip = 4 }
				}
				modifier = {
					factor = 1.2
					NOT = { dip = 3 }
				}
				modifier = {
					factor = 1.2
					NOT = { dip = 2 }
				}
				add_dip_power = -25
			}
		}
		custom_tooltip = pf_feast_results_depend_on_ruler_skills_dip_tt
	}
	option = {		# Ignore them
		name = pf_feast.2.c
		add_prestige = -10
	}
	option = {		# This is the kind of atmosphere I want in my court!
		name = pf_feast.2.e
		trigger = {
			ruler_has_personality = scholar_personality
		}
		highlight = yes
		add_dip_power = 50
	} 
}
