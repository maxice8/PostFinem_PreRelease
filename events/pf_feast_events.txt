		clr_country_flag = is_in_feast
	}

		name = "pf_feast.1.a"
		trigger = {
			government_rank = 1
		}

			limit = {
				NOT = { has_government_attribute = enables_discounted_host_feast }

				name = "pf_feast_reputation_boost"
				duration = -1
				desc = "until the end of the feast"
			}

			country_event = {
				id = pf_feast.10

	option = {
		name = "pf_feast.1.b"

			government_rank = 2
		}
		highlight = yes
		if = {

		}

					factor = 1.2
