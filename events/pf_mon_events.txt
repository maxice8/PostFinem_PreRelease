        }

province_event = { # Blessing of Zeus
    id = pf_mon_events.4
    title = pf_mon_events.4.title
    desc = pf_mon_events.4.desc
    picture = NORSE_TEMPLE_eventPicture
    goto = ROOT
    mean_time_to_happen = {months = 6} # This is in concurrence with the flag cooldown
    
    trigger = {
        has_great_project = {
            type = hall_of_pantheons
            tier = 1
        }
        province_is_or_accepts_religion = {religion = hellenic}
        if = { # Reduced cooldown if the monument is tier 2
            limit = {
                has_great_project = {
                    type = hall_of_pantheons
                    tier = 2
                }
            }
            OR = {
                NOT = {has_province_flag = pf_mon_hop_blessing_flag}
                had_province_flag = {
                    flag = pf_mon_hop_blessing_flag
                    days = 2555
                }
                AND = { # Reduced cooldown if you have <0 stability
                    had_province_flag = {
                        flag = pf_mon_hop_blessing_flag
                        days = 1788
                    }
                    NOT = {owner = {stability = 0}}
                }
            }
        }
        else = {
            OR = {
                NOT = {has_province_flag = pf_mon_hop_blessing_flag}
                had_province_flag = {

                        flag = pf_mon_hop_blessing_flag
                        days = 2555
                    }
                    NOT = {owner = {stability = 0}}
                }
            }
        }
    }

    option = {
        name = pf_mon_events_we_are_truly_blessed
        trigger = {
            NOT = {

                    type = hall_of_pantheons

            factor = 1

            set_country_flag = pf_mon_hop_recieved_blessing # Used to track blessing in other places.

            add_devotion = 15
            add_horde_unity = 15
            add_meritocracy = 15
        }
    }
    option = {
        name = pf_mon_events_we_are_truly_blessed
