namespace = pf_mon_events

province_event = { # Blessing of Zeus
    id = pf_mon_events.1
    title = pf_mon_events.1.title
    desc = pf_mon_events.1.desc
    picture = NORSE_TEMPLE_eventPicture
    goto = ROOT

    mean_time_to_happen = { # This is in concurrence with the flag cooldown
        months = 6
    }

    trigger = {
        has_great_project = {
            type = hall_of_pantheons
            tier = 1
        }
        province_is_or_accepts_religion = {
            religion = hellenic
        }
        OR = {
            NOT = {
                has_province_flag = pf_mon_hop_blessing_flag
            }
            had_province_flag = {
                flag = pf_mon_hop_blessing_flag
                days = 1460
            }
            AND = { # Reduced cooldown if you have <0 stability
                had_province_flag = {
                    flag = pf_mon_hop_blessing_flag
                    days = 1095
                }
                NOT = {
                    owner = {
                        stability = 0
                    }
                }
            }
        }
    }

    option = {
        name = pf_mon_events_we_are_truly_blessed
        trigger = {
        }
        ai_chance = {
            factor = 1
        }
        set_province_flag = pf_mon_hop_blessing_flag
        owner = {
            add_legitimacy = 15
            add_republican_tradition = 5
            add_devotion = 15
            add_horde_unity = 15
            add_meritocracy = 15
        }
    }
}
