	trigger = {  }

	fire_only_once = yes
	
	is_triggered_only = yes
	# mean_time_to_happen = {}

	immediate = {  
		hidden_effect = {
			add_stability = -1
			define_general = {

				fire = 5
				shock = 4
				manuever = 5
				siege = 2
			}

					type = leadered_peasant_rebels
					size = 2
				}
			}			
			random_owned_province = {
				spawn_rebels = {
					type = leadered_peasant_rebels
					size = 2
				}
			}			
			random_owned_province = {
				spawn_rebels = {
					type = particularist_rebels
					size = 3
				}
			}
		}
	}

	option = { #Fuck it, we ball
		name = pf_mac_disaster.1.a
		# goto = 
		ai_chance = { factor = 1 }

		tooltip = {	add_stability = -1 }
		kill_heir = { allow_new_heir = no }
		exile_ruler_as = { name = exiled_mac_ruler }
		custom_tooltip = empty_line_tt
		if = {
			limit = {
				has_leader = "Nico Samaras"
			}
			define_leader_to_ruler = {
				name = "Nico Samaras"
				adm = 1
				dip = 1
				mil = 3
				# fixed = no
				claim = 50
			}
		}
		else = {
			define_general = {
				name = "Nico Samaras"
				fire = 5
				shock = 4
				manuever = 5
				siege = 2
			}
			define_leader_to_ruler = {
				name = "Nico Samaras"
				adm = 1
				dip = 1
				mil = 2
				# fixed = no
				claim = 50
			}
		}
