# Eyl considers entry
country_event = {
	id = incidents_eyl_join_empire.1
	title = "incidents_eyl_join_empire.1.t"
	desc = "incidents_eyl_join_empire.1.desc"
	picture = HRE_eventPicture
	
	major = yes
	fire_only_once = yes
	
	trigger = {
		is_year = 1446
		EYL = {
			has_reform = pirate_republic_reform
		}
	}

	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.75
			alliance_with = emperor
		}
		modifier = {
			factor = 0.75
			has_opinion = {
				who = emperor
				value = 100
			}
		}
	}
	
	option = {
		name = "incidents_eyl_join_empire.1.a"
		ai_chance = {
			factor = 10
			modifier = {
				factor = 1.5
				has_opinion = {
					who = emperor
					value = 100
				}
			}
			modifier = {
				factor = 1.25
				NOT = { is_rival = emperor }
			}
			modifier = {
				factor = 1.25
				NOT = { emperor = { is_rival = ROOT } }
			}
		}
		custom_tooltip = eyl_start_incident_tt
		hidden_effect = {
			set_imperial_incident = incident_eyl_wants_to_join
		}
	}
	option = {
		name = "incidents_eyl_join_empire.1.b"
		ai_chance = {
			factor = 10
			modifier = {
				factor = 1.5
				NOT = {
					has_opinion = {
						who = emperor
						value = 0
					}
				}
			}
			modifier = {
				factor = 1.25
				is_rival = emperor
			}
			modifier = {
				factor = 1.25
				emperor = { is_rival = ROOT }
			}
		}
		set_country_flag = eyl_refused_empire
		custom_tooltip = eyl_refuse_to_join_empire_tt
	}
}
