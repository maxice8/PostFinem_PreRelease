on_siege_won_province = {
	if = {
		limit = {
			controller = {
				is_in_war = {
					casus_belli = cb_raid
					attacker_leader = THIS
				}
			}
		}
		ROOT = {
			export_to_variable = {
				which = RaidFromProv
				value = development
				who = ROOT
			}
			# space for modifiers
			while = {
				limit = {
					check_variable = {
						which = RaidFromProv
						value = 1
					}
				}
				subtract_variable = {
					which = RaidFromProv
					value = 1
				}
				controller = {
					add_loot = {
						amount = 1
					}
				}
			}
			if = {
				limit = {
					has_tax_building_trigger = yes
					controller = {
						has_government_attribute = raids_give_money_from_temples
					}
				}
				if = {
					limit = { has_building = temple }
					controller = {
						add_loot = {
							amount = 5
						}
					}
				}
				else_if = {
					limit = { has_building = cathedral }
					controller = {
						add_loot = {
							amount = 10
						}
					}
				}
			}
		}
		if = {
			limit = {
				controlled_by = KSB
				area = vorpommern_area
			}
			save_event_target_as = sieged_polabian_province
			controller = {
				country_event = {
					id = pf_ksb_events.3
				}
			}
		}
	}
}

on_province_culture_changed = {
	if = {
		limit = {
			owner = {
				has_reform = enlightened_norse_kingdom
			}
			has_owner_accepted_culture = no
			has_owner_culture = no
			culture_group = owner
		}
		add_province_modifier = {
			name = nld_tolerant_culture_group
			duration = -1
		}
	}
	else_if = {
		limit = {
			owner = {
				has_reform = enlightened_norse_kingdom
			}
			has_owner_accepted_culture = no
			has_owner_culture = no
			NOT = {
				culture_group = owner
			}
		}
		add_province_modifier = {
			name = nld_tolerant_culture
			duration = -1
		}
	}

	if = {
		limit = {
			has_province_modifier = nld_tolerant_culture_group
			NOT = {
				OR = {
					AND = {
						has_owner_accepted_culture = no
						has_owner_culture = no
						culture_group = owner
					}
					owner = {
						has_reform = enlightened_norse_kingdom
					}
				}
			}
			
		}
		remove_province_modifier = nld_tolerant_culture_group
	}
	else_if = {
		limit = { 
			has_province_modifier = nld_tolerant_culture
			NOT = {
				OR = {
					AND = {
						has_owner_accepted_culture = no
						has_owner_culture = no
						NOT = {
							culture_group = owner
						}
					}
					owner = {
						has_reform = enlightened_norse_kingdom
					}
				}
			}
		}
		remove_province_modifier = nld_tolerant_culture
	}
}

on_province_owner_change = {
	if = {
		limit = {
			owner = {
				OR = {
					has_country_flag = pf_sami_herding_flag
					AND = {
						colonial_parent = {
							has_country_flag = pf_sami_herding_flag
							mission_completed = pf_nld_northern_colonies
						}
						capital_scope = {
							colonial_region = colonial_alaska
						}
					}
					AND = {
						colonial_parent = {
							has_country_flag = pf_sami_herding_flag
							mission_completed = pf_nld_a_bit_of_truth_in_legends
						}
						capital_scope = {
							colonial_region = colonial_canada
						}
					}
					AND = {
						colonial_parent = {
							has_country_flag = pf_sami_herding_flag
							mission_completed = pf_nld_spread_westwards
						}
						capital_scope = {
							colonial_region = colonial_louisiana
						}
					}
					AND = {
						colonial_parent = {
							has_country_flag = pf_sami_herding_flag
							mission_completed = pf_nld_new_sweden
						}
						capital_scope = {
							colonial_region = colonial_eastern_america
						}
					}
				}
			}
			culture = sapmi
			trade_goods = fur
		}
		add_province_modifier = {
			name = sami_herding
			duration = -1
		}
	}
	else_if = {
		limit = {
			has_province_modifier = sami_herding
		}
		remove_province_modifier = sami_herding
	}

	if = {
		limit = {
			owner = {
				OR = {
					has_country_flag = pf_sami_herding_flag
					AND = {
						colonial_parent = {
							has_country_flag = pf_sami_herding_flag
							mission_completed = pf_nld_northern_colonies
						}
						capital_scope = {
							colonial_region = colonial_alaska
						}
					}
					AND = {
						colonial_parent = {
							has_country_flag = pf_sami_herding_flag
							mission_completed = pf_nld_a_bit_of_truth_in_legends
						}
						capital_scope = {
							colonial_region = colonial_canada
						}
					}
				}
			}
			has_climate = arctic
		}
		add_province_modifier = {
			name = swe_arctic_expansion
			duration = -1
		}
	}

	if = {
		limit = {
			owner = {
				NOT = {
					has_country_flag = pf_arctic_expansion_flag
				}
			}
			has_province_modifier = swe_arctic_expansion
		}
		remove_province_modifier = swe_arctic_expansion
	}

	if = {
		limit = {
			owner = {
				has_reform = enlightened_norse_kingdom
			}
			has_owner_accepted_culture = no
			has_owner_culture = no
			culture_group = owner
		}
		add_province_modifier = {
			name = nld_tolerant_culture_group
			duration = -1
		}
	}
	 if = {
		limit = {
			owner = {
				has_reform = enlightened_norse_kingdom
			}
			has_owner_accepted_culture = no
			has_owner_culture = no
			NOT = {
				culture_group = owner
			}
		}
		add_province_modifier = {
			name = nld_tolerant_culture
			duration = -1
		}
	}
}

on_admiral_recruited = {
	if = {
		limit = {
			has_country_flag = pf_sar_recruiting_admirals_gives_tradition
		}
		add_navy_tradition = 2
	}
}
