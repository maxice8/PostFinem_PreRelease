						alliance_with = FROM

					hidden_progressive_opinion_for_scholar_trigger = yes
				}
				if = {
					limit = {
						ai = yes
					}
					NOT = { has_country_modifier = has_invited_scholar_recently }
				}
				NOT = { has_country_modifier = orthodox_school_modifier }
			}
