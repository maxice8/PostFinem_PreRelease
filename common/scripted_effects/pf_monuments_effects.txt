stab_for_tributaries = {
	custom_tooltip = stab_for_tributaries_tt
	hidden_effect = {
		every_subject_country = {
			if = {
				limit = {
					is_subject_of_type = tributary_state
				}
				overlord = {
					add_stability = 1
				}
			}
		}
	}
}

grand_temple_to_tanit_1 = {
	custom_tooltip = grand_temple_to_tanit_1_tt
	hidden_effect = {
		every_capital_province = {
			if = {
				limit = {
					region = france_region
					owner = {
						NOT = {
							religion = punic_religion
						}
					}
				}
				owner = {
					add_opinion  = {
						who = ROOT
						modifier = opinion_grand_temple_to_tanit
					}
					reverse_add_opinion  = {
						who = ROOT
						modifier = opinion_grand_temple_to_tanit
					}
				}
			}
		}
	}
}
