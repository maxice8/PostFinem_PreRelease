pf_mon_sod_stab_for_tributaries = {
	custom_tooltip = pf_mon_sod_stab_for_tributaries_tt
	hidden_effect = {
		every_subject_country = {
			if = {
				limit = {
					is_subject_of_type = tributary_state
				}
				overlord = {
					add_stability = 1
				}
			}
		}
	}
}

pf_mon_gtt_effect_1 = {
	custom_tooltip = pf_mon_gtt_effect_1_tt
	hidden_effect = {
		every_capital_province = {
			if = {
				limit = {
					region = france_region
					owner = {
						NOT = {
							religion = punic_religion
						}
					}
				}
				owner = {
					add_opinion  = {
						who = ROOT
						modifier = pf_mon_gtt_opinion_from_upgrade
					}
					reverse_add_opinion  = {
						who = ROOT
						modifier = pf_mon_gtt_opinion_from_upgrade
					}
				}
			}
		}
	}
}
