pf_bav_1 = {
	slot = 1
	generic = no
	ai = yes
	has_country_shield = yes
	potential = { tag = BAV } 

	bav_southern_lords = {
		icon = mission_have_two_subjects
		position = 1
		provinces_to_highlight = {  } 

		trigger = {
			OR = {
				army_size_percentage = 1
				AND = {
					ai = yes
					army_size_percentage = 0.75
					num_of_generals = 1
				}
			}
		}

		effect = {
			upper_bavaria_area = { add_permanent_claim = ROOT } 
			upper_swabia_area = { add_permanent_claim = ROOT } 
			tirol_area = { add_permanent_claim = ROOT } 
		}
	}

	bav_capital = {
		icon = mission_high_income
		position = 3
		required_missions = { bav_southern_lords } 
		provinces_to_highlight = { province_id = 64 } 

		trigger = {
			64 = {
				owned_by = ROOT
				development = 12
			}
		}

		effect = {
			64 = {
				if = {
					limit = { has_building = courthouse } 
					add_base_tax = 1
					add_base_production = 1
					add_base_manpower = 1
				}
				else = { add_building = courthouse } 
			}
		}
	}
}

	bav_marklander = {
		icon = the_imperial_border
		position = 3

		}
		provinces_to_highlight = {
			OR = {
				area = erzgebirge_area
				area = bohemia_area
			}
		}

		trigger = {

					}
				}
				MVA = { is_subject_of = ROOT } 
			}
		}

		effect = {
			if = {
				limit = { MVA = { is_subject_of = ROOT }  }

			on_change_tag_effect = yes

			swap_non_generic_missions = yes
			remove_non_electors_emperors_from_empire_effect = yes

				limit = { has_custom_ideas = no } 
				country_event = { id = ideagroups.1 } 
			}
			set_country_flag = formed_markland_flag
		}
	}
