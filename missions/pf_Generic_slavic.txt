	sev_rising_power = {
		icon = mission_assemble_an_army
		position = 1
		provinces_to_highlight = {  
			
		} 

		trigger = {  
			if = {
				limit = {
					ai = no
				}
				army_size_percentage = 1
				num_of_generals = 1
			}
			else = {
				army_size_percentage = 0.75
			}

		} 

		effect = {  
			poland_region = {
				add_claim = ROOT
			}
		} 
	}

	sev_conquer_severoslavia = {
		icon = mission_monarch_in_throne_room
		position = 2
		required_missions = { sev_rising_power } 
		provinces_to_highlight = { 
			region = poland_region
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		 } 

		trigger = {  
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 25
				region = poland_region
			}
		} 

		effect = {  
			add_prestige_or_monarch_power = { amount = 25 }
			custom_tooltip = empty_line_tt
			custom_tooltip = ksb_can_form_severoslavia_tt
			set_country_flag = can_form_severoslavia_earlier_flag
			south_slavia_region = {
				add_claim = ROOT
			}
			north_german_region = {
				add_claim = ROOT
			}
		} 
	}

	sev_push_south = {
		icon = danubian_power
		position = 3
		required_missions = { sev_conquer_severoslavia }

		}

			}

	sev_lord_of_slavia = {
		icon = mission_empire
		position = 4
		required_missions = { sev_push_south } 
		provinces_to_highlight = {  } 

		trigger = {  
			tag = SLV
		} 

		effect = {  
			add_country_modifier = {
				name = ksb_lord_of_all_slavs
				duration = -1
			}

				add_claim = ROOT
			}
			russia_region = {
				add_claim = ROOT
			}
			baltic_region = {
				add_claim = ROOT
			}
		} 
	}

	sev_conquer_rus = {
		icon = mission_steppe_warriors
		position = 5
		required_missions = { sev_lord_of_slavia } 
		provinces_to_highlight = {  
			OR ={
				region = russia_region
				region = ruthenia_region
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		} 

		trigger = {  
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 30
				OR = {
					region = ruthenia_region
					region = russia_region
				}
			}
			280 = {
				owned_by = ROOT
				if = {
					limit = {
						ROOT = {
							ai = no
						}
					}
					development = 20
				}
			}
		} 

		effect = {  
			country_event = {
				id = pf_ksb_events.21
			}
			custom_tooltip = empty_line_tt
			280 = {
				change_culture = rus_new
				custom_tooltip = empty_line_tt
				add_province_modifier = {
					name = "ksb_rus_capital"
					duration = -1
				}
			}
			# custom_tooltip = ksb_rus_culture_converting_tt
			# set_country_flag = ksb_converting_rus_flag
		} 
	}
