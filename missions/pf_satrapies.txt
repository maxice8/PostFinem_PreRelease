pf_afg_5 = {
	slot = 5
	generic = no
	ai = yes
	has_country_shield = yes

	potential = {
		tag = AFG
	}

	pf_afg_satrapy_expansion_plan = {
		icon = mission_unknown_mission
		position = 3

		required_missions = { }

		provinces_to_highlight = {
			province_group = pf_afg_expansion_claims
			NOT = { owned_by = ROOT }
		}

		trigger = {
			custom_trigger_tooltip = {
				tooltip = pf_satrapy_expansion_plan_missions_tt
				pf_afg_expansion_claims = {
					type = ALL
					owned_by = ROOT
				}
			}
		}

		effect = {
			add_country_modifier = {
				name = pf_satrapy_expansion_plan_complete
				duration = -1
				desc = pf_satrapy_while_subject_tt
			}
			set_country_flag = pf_per_satrapy_finished
		}
	}
}
