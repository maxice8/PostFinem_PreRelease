	slot = 1
	generic = no
	ai = yes
	has_country_shield = yes
	potential = { tag = MKM } 

	mkm_bordermania = {
		icon = mission_colonial
		position = 1
		provinces_to_highlight = { culture = marcomanni } 

		trigger = {
			owns = 64
			64 = { development = 10 } 
			num_of_owned_provinces_with = {
				value = 8
				culture = marcomanni
				development = 10
			}
		}

		effect = {
			64 = {
				if = {
					limit = { province_has_center_of_trade_of_level = 1 } 
					add_base_tax = 1
					add_base_production = 1
					add_base_manpower = 1
				}
				else = { center_of_trade = 1 } 
			}
		}
	}

	mkm_home_sweet_home = {
		icon = mission_conquer_50_development
		position = 3

		provinces_to_highlight = { province_id = 64 } 

		trigger = {
			64 = {
				owned_by = ROOT
				development = 15
				province_has_center_of_trade_of_level = 2
			}
		}

		effect = {
			64 = {
				add_province_modifier = {
					name = "mkm_salty"
					duration = -1
				}
			}
		}
	}
}

			}

		trigger = {
			num_of_provinces_owned_or_owned_by_subjects_with = {
				value = 8
				OR = {
					area = moravia_area
					area = slovakia_area
				}
			}
		}

		effect = {
			add_country_modifier = {
				name = "mkm_the_long_climb"
				duration = 9125
			}
		}
	}

	mkm_brotherhood = {
		icon = unite_the_realm
		position = 3
		required_missions = {
			mkm_ostholm
			mkm_bergaland
			mkm_vindheim
		}
		provinces_to_highlight = {  } 

		trigger = {
			accepted_culture = fyrnir
			accepted_culture = quadi
			accepted_culture = eldrun
			OR = {
				accepted_culture = haldyr
				dip_power = 100
			}
		}

		effect = {
			if = { limit = { accepted_culture = haldyr }  } 
			else = { add_dip_power = -100 } 
			if = {
				limit = { has_country_modifier = mkm_ostgermanic_culture } 
				remove_country_modifier = mkm_ostgermanic_culture
			}
			else = { custom_tooltip = CT_MKM_1 } 
			add_country_modifier = {
				name = "mkm_ostgermanic_culture_2"
				duration = -1
			}
		}
	}

	mkm_ost_germania = {

		provinces_to_highlight = { region = carpathia_region } 

		trigger = {
			carpathia_region = {

			on_change_tag_effect = yes
			swap_non_generic_missions = yes
			remove_non_electors_emperors_from_empire_effect = yes
			set_government_rank = 2

				name = "centralization_modifier"
				duration = 7300
			}
			add_prestige = 25
			if = {
				limit = { has_custom_ideas = no } 
				country_event = { id = ideagroups.1 } 
			}
		}
	}
}
